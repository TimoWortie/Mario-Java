		settingsclose = new JLabel();
		settingsclose.setIcon(new ImageIcon(getClass().getResource("/resource/close_default.png")));
		settingsclose.setBounds(1030,-15,200,100);
		settingsclose.addMouseListener(new MouseAdapter() {
									
					boolean sgHover = false;
							
					public void mouseEntered(MouseEvent evt) {
						settingsclose.setIcon(new ImageIcon(getClass().getResource("/resource/close_hover.png")));
						sgHover = true;
					}
					public void mouseExited(MouseEvent evt) {
						settingsclose.setIcon(new ImageIcon(getClass().getResource("/resource/close_default.png")));
						sgHover = false;
					}
					public void mousePressed(MouseEvent evt) {
						settingsclose.setIcon(new ImageIcon(getClass().getResource("/resource/close_pressed.png")));
						
						settingsframe.dispose();

							
					}
					public void mouseReleased(MouseEvent evt) {
						if (sgHover == true) {
							settingsclose.setIcon(new ImageIcon(getClass().getResource("/resource/close_hover.png")));
							
							}
					}
		});
		settingslabel.add(settingsclose);
